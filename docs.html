Impulse Function Documentation

Complete reference for all available functions and aliases

Custom Impulse Functions

+
Impulse.getpcname()
Returns the name of the current PC.
local pcName = Impulse.getpcname()
print("PC Name:", pcName)
Impulse.gethwid()
Returns the hardware ID of the current machine.
local hwid = Impulse.gethwid()
print("HWID:", hwid)
Impulse.info()
Returns general information about the Impulse executor.
local info = Impulse.info()
print("Info:", info)
Impulse.isimpulse()
Returns true if the current executor is Impulse.
if Impulse.isimpulse() then
  print("Using Impulse!")
end
Impulse.identify()
Returns identification information about the executor.
local ident = Impulse.identify()
print("Identification:", ident)
Impulse.isvirtualmachine()
Returns true if the current machine is a virtual machine.
if Impulse.isvirtualmachine() then
  print("Running in VM!")
end
Impulse.isdebuggerpresent()
Returns true if a debugger is attached to the process.
if Impulse.isdebuggerpresent() then
  print("Debugger detected!")
end
Impulse.tip()
Returns a random tip about using Impulse.
local tip = Impulse.tip()
print("Tip:", tip)
Impulse.version()
Returns the current version of Impulse.
local version = Impulse.version()
print("Version:", version)
Impulse.detections()
Returns information about current detections.
local detections = Impulse.detections()
print("Detections:", detections)
Impulse.discord() / Impulse.invite() Aliases: invite
Returns the Discord invite link for Impulse.
local discordLink = Impulse.discord()
print("Join our Discord:", discordLink)

Input Functions

+
isrbxactive() / isgameactive() / iswindowactive() Aliases: isgameactive, iswindowactive
Returns true if the Roblox window is active.
if isrbxactive() then
  print("Roblox is focused")
end
keypress(keyCode) / keytap(keyCode) / keyrelease(keyCode) Aliases: keytap, keyrelease
Simulates key press/tap/release. KeyCode is the virtual key code.
keypress(0x41) -- 'A' key
wait(0.1)
keyrelease(0x41)
mouse1click() / mouse1press() / mouse1release() Aliases: mouse1press, mouse1release
Simulates left mouse button click/press/release.
mouse1press()
wait(0.1)
mouse1release()
mouse2click() / mouse2press() / mouse2release() Aliases: mouse2press, mouse2release
Simulates right mouse button click/press/release.
mouse2click()
mousemoveabs(x, y) / mousemoverel(x, y) Aliases: mousemoverel
Moves mouse to absolute screen coordinates or relative to current position.
mousemoveabs(100, 100) -- move to 100,100
mousemoverel(50, 0) -- move right 50 pixels
mousescroll(amount)
Simulates mouse wheel scrolling. Positive for up, negative for down.
mousescroll(5) -- scroll up
mousescroll(-5) -- scroll down

fireclickdetector(clickDetector[, distance])
Fires the specified ClickDetector. Optional distance parameter.
local detector = game.Workspace.Part.ClickDetector
fireclickdetector(detector, 10)

Fires a touch interest between two parts. Touch is 0 for begin and 1 for end.
local part1 = game.Workspace.Part1
local part2 = game.Workspace.Part2
firetouchinterest(part1, part2, 0) -- begin touch
wait(1)
firetouchinterest(part1, part2, 1) -- end touch

Fires the specified ProximityPrompt. Optional distance parameter.
local prompt = game.Workspace.Part.ProximityPrompt

gethui()
Returns the CoreGui service (same as game:GetService("CoreGui")).
local coreGui = gethui()
print(coreGui:GetChildren())
getcallbackvalue(callback)
Gets the value returned by a callback function.
local callback = function() return "Hello" end
local value = getcallbackvalue(callback)

cloneref(instance)
Creates a clone of the reference that bypasses filtering.
local part = game.Workspace.Part

compareinstances(instance1, instance2)
Compares two instances to see if they reference the same object.

local part2 = game.Workspace.Part2
print(compareinstances(part1, part2)) -- false

Metatable Functions

+
getrawmetatable(table)
Returns the metatable of the given table, bypassing any __metatable protection.
local mt = getrawmetatable(game)
print(mt)
setrawmetatable(table, metatable)
Sets the metatable of the given table, bypassing any __metatable protection.
local t = {}
local mt = {}
setrawmetatable(t, mt)
isreadonly(table)
Returns true if the table is read-only.
print(isreadonly(game))
setreadonly(table, readonly)
Sets whether a table is read-only.
local t = {}
setreadonly(t, true)
getnamecallmethod()
Returns the name of the current namecall method.
local oldnc; oldnc = hookmetamethod(game, "__namecall", function(...)
  local method = getnamecallmethod()
  print("Namecall:", method)
  return oldnc(...)
end)
hookmetamethod(table, method, hook) / hookfunction(func, hook) Aliases: hookfunction
Hooks a metamethod or function to replace its behavior.
-- Hook __index metamethod
local old; old = hookmetamethod(game, "__index", function(t, k)
  print("Indexing:", k)
  return old(t, k)
end)

-- Hook a function
local oldPrint = hookfunction(print, function(...)
  print("Intercepted print")
  return oldPrint(...)
end)
checkcaller()
Returns true if the current caller is from the executor.
if checkcaller() then
  print("Called from executor")
end
clonefunction(func)
Creates a clone of the given function.
local function original() print("Original") end
local clone = clonefunction(original)
clone() -- "Original"
newcclosure(func)
Converts a function to a C closure.
local function myFunc() end
local cfunc = newcclosure(myFunc)
iscclosure(func)
Returns true if the function is a C closure.
local function myFunc() end
print(iscclosure(myFunc)) -- false
print(iscclosure(newcclosure(myFunc))) -- true
isexecutorclosure(func)
Returns true if the function is an executor closure.
local function myFunc() end
print(isexecutorclosure(myFunc)) -- false
islclosure(func)
Returns true if the function is a Lua closure.
local function myFunc() end
print(islclosure(myFunc)) -- true
print(islclosure(newcclosure(myFunc))) -- false
getscripts()
Returns a table of all scripts in the game.
local scripts = getscripts()
for i, script in pairs(scripts) do

Returns true if the property of the instance is scriptable.

Debug Functions

+
getreg() / getregistry() Aliases: getregistry
Returns the Lua registry table.
local reg = getreg()
for k, v in pairs(reg) do
  -- Do something with registry entries
end
getproto(func, index)
Gets the prototype at the given index from a function.
local function test() end
local proto = getproto(test, 0)
getprotos(func)
Gets all prototypes from a function.
local function test() end
local protos = getprotos(test)
for i, proto in ipairs(protos) do
  -- Do something with prototypes
end
getstack([level])
Gets the call stack at the given level (default 0).
local function test()
  local stack = getstack(1)
  print(stack)
end
test()
getinfo([func[, what]])
Gets information about a function or the current function if none is provided.
local function test() end
local info = getinfo(test)
print(info.name, info.linedefined, info.lastlinedefined)
getupvalue(func, index)
Gets the upvalue at the given index from a function.
local upval = "hello"
local function test() print(upval) end
local name, value = getupvalue(test, 1)
print(name, value) -- "upval", "hello"
getupvalues(func)
Gets all upvalues from a function.
local upval1 = "hello"
local upval2 = "world"
local function test() print(upval1, upval2) end
local upvalues = getupvalues(test)
for name, value in pairs(upvalues) do
  print(name, value)
end
getconstant(func, index)
Gets the constant at the given index from a function.
local function test() print("constant") end
local constant = getconstant(test, 1)
print(constant) -- "constant"
getconstants(func)
Gets all constants from a function.
local function test() print("hello", "world") end
local constants = getconstants(test)
for i, constant in ipairs(constants) do
  print(constant)
end
setconstant(func, index, value)
Sets the constant at the given index in a function.
local function test() print("hello") end
setconstant(test, 1, "world")
test() -- prints "world"
setstack(level, value)
Sets the call stack at the given level.
-- Advanced usage only
setupvalue(func, index, value)
Sets the upvalue at the given index in a function.
local upval = "hello"
local function test() print(upval) end
setupvalue(test, 1, "world")
test() -- prints "world"

Cryptography Functions

+
base64_encode(data) / base64_decode(data) Aliases: base64_decode
Encodes/decodes data using Base64.
local encoded = base64_encode("Hello")
local decoded = base64_decode(encoded)
print(decoded) -- "Hello"
base64.encode(data) / base64.decode(data) Aliases: base64.decode
Alternative syntax for Base64 encoding/decoding.
local encoded = base64.encode("World")
local decoded = base64.decode(encoded)
crypt.encrypt(data, key) / crypt.decrypt(data, key) Aliases: crypt.decrypt
Encrypts/decrypts data using the specified key.
local key = "secretkey"
local encrypted = crypt.encrypt("Data", key)
local decrypted = crypt.decrypt(encrypted, key)
print(decrypted) -- "Data"
crypt.generatebytes(length)
Generates random bytes of the specified length.
local randomBytes = crypt.generatebytes(16)
print(randomBytes)
crypt.generatekey()
Generates a random encryption key.
local key = crypt.generatekey()
print(key)
crypt.hash(data, algorithm)
Hashes data using the specified algorithm (e.g., "md5", "sha1", "sha256").
local hash = crypt.hash("password", "sha256")
print(hash)
